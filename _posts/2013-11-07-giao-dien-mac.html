[INSTALL]

wget -O mac-cursors.zip http://goo.gl/xh52


//Tải gói mac-cursors.zip về.

sudo unzip mac-cursors.zip -d /usr/share/icons/


cd /usr/share/icons/mac-cursors

./install-mac-cursors.sh
